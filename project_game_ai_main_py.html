142 def determine_strategy(self, tiles_136, meld_tile=None): 

143 # for already opened hand we don't need to give up on selected strategy 

144 if self.player.is_open_hand and self.current_strategy: 

145 return False 

146 

147 old_strategy = self.current_strategy 

148 self.current_strategy = None 

149 

150 # order is important, we add strategies with the highest priority first 

151 strategies = [] 

152 

153 if self.player.table.has_open_tanyao: 

154 strategies.append(TanyaoStrategy(BaseStrategy.TANYAO, self.player)) 

155 

156 strategies.append(YakuhaiStrategy(BaseStrategy.YAKUHAI, self.player)) 

157 strategies.append(HonitsuStrategy(BaseStrategy.HONITSU, self.player)) 

158 strategies.append(ChinitsuStrategy(BaseStrategy.CHINITSU, self.player)) 

159 

160 strategies.append(FormalTempaiStrategy(BaseStrategy.FORMAL_TEMPAI, self.player)) 

161 strategies.append(CommonOpenTempaiStrategy(BaseStrategy.COMMON_OPEN_TEMPAI, self.player)) 

162 

163 for strategy in strategies: 

164 if strategy.should_activate_strategy(tiles_136, meld_tile=meld_tile): 

165 self.current_strategy = strategy 

166 break 

167 

168 if self.current_strategy and (not old_strategy or self.current_strategy.type != old_strategy.type): 

169 self.player.logger.debug( 

170 log.STRATEGY_ACTIVATE, 

171 context=self.current_strategy, 

172 ) 

173 

174 if not self.current_strategy and old_strategy: 

175 self.player.logger.debug(log.STRATEGY_DROP, context=old_strategy) 

176 

177 return self.current_strategy and True or False

292 def estimate_weighted_mean_hand_value(self, discard_option): 

293 weighted_hand_cost = 0 

294 number_of_tiles = 0 

295 for waiting in discard_option.waiting: 

296 tiles = self.player.tiles[:] 

297 tiles.remove(discard_option.find_tile_in_hand(self.player.closed_hand)) 

298 

299 hand_cost = self.estimate_hand_value_or_get_from_cache( 

300 waiting, tiles=tiles, call_riichi=not self.player.is_open_hand, is_tsumo=True 

301 ) 

302 

303 if not hand_cost.cost: 

304 continue 

305 

306 weighted_hand_cost += ( 

307 hand_cost.cost["main"] + 2 * hand_cost.cost["additional"] 

308 ) * discard_option.wait_to_ukeire[waiting] 

309 number_of_tiles += discard_option.wait_to_ukeire[waiting] 

310 

311 cost = number_of_tiles and int(weighted_hand_cost / number_of_tiles) or 0 

312 

313 # we are karaten, or we don't have yaku 

314 # in that case let's add possible tempai cost 

315 if cost == 0 and self.player.round_step > 12: 

316 cost = 1000 

317 

318 if self.player.round_step > 15 and cost < 2500: 

319 cost = 2500 

320 

321 return cost
